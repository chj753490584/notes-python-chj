###########Function
def Sum_up(x, y):
    return x + y

Prod = lambda x, y: x * y

def Fib(n):
    fib = [1, 1]
    for k in range(n):
        a = fib[-2] + fib[-1]
        fib.append(a)
    return fib

z = Sum_up(2, 3)
print z
y = Prod(2, 3)
print y
print Fib(10)
